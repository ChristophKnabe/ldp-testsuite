072        public void generateReport(List<XmlSuite> xmlSuites, List<ISuite> suites,
073                                                           String outputDirectory) {
074                try {
075                        for (ISuite suite : suites) {
076                                html = new HtmlCanvas();
077                                html.html().head();
078
079                                writeCss();
080                                html.title().content(LdpTestSuite.NAME + " Report")._head()
081                                                .body();
082                                html.h1().content(LdpTestSuite.NAME + " Summary");
083
084                                // Getting the results for the said suite
085                                Map<String, ISuiteResult> suiteResults = suite.getResults();
086
087                                for (ISuiteResult sr : suiteResults.values()) {
088
089                                        ITestContext tc = sr.getTestContext();
090                                        passedTests = tc.getPassedTests();
091                                        failedTests = tc.getFailedTests();
092                                        skippedTests = tc.getSkippedTests();
093                                }
094                                
095                                // Initialize variables for charts
096                                passClasses = getClasses(passedTests);
097                                failClasses = getClasses(failedTests);
098                                skipClasses = getClasses(skippedTests);
099                                
100                                html.h2().content("Overall Coverage Bar Charts");
101                                
102                                html.span(class_("chartStart"));
103                                html.label(class_("label")).b().write("Test Results by Specification Requirement")._b()._label();
104                                html.div(class_("barChart").id("overallChart1"))._div(); // svg chart
105                                writePassFailLegend();
106                                
107                                html._span();
108                                
109                                html.span(class_("chartStart"));
110                                html.label(class_("label")).b().write("Test Results by Test Class")._b()._label();
111                                html.div(class_("barChart").id("resourcesChart"))._div();
112                                writeTestClassLegend();
113                                html._span();
114                                
115                                html.br();
116                                generateOverallSummaryReport(suites, "summary");
117                                displayGroupsInfo(suites);
118                                displayMethodsSummary(suites);
119                                toTop();
120                                generateMethodDetails(suites);
121
122                                html.script().content(
123                                                StringResource.get("/raphael/raphael-min.js"), NO_ESCAPE);
124                                html.script().content(
125                                                StringResource.get("/prototype/prototype.js"), NO_ESCAPE);
126                                html.script().content(
127                                                StringResource.get("/grafico/grafico-min.js"), NO_ESCAPE);
128                                
129                                writeOverallBarChart();
130                                writeResourcesBarChart();
131                
132                                html.write(graphs.toString(), NO_ESCAPE);
133                                html._body()._html();
134
135                                // send html to a file
136                                createWriter("report", html.toHtml());
137
138                                Files.copy(getClass().getResourceAsStream("/testng-reports.css"),
139                                                new File(outputDirectory, "testng-reports.css").toPath(),
140                                                StandardCopyOption.REPLACE_EXISTING);
141                        }
142
143                } catch (IOException e) {
144                        e.printStackTrace();
145                }
146        }

154        private void createWriter(String directory, String output) {
155                BufferedWriter writer = null;
156                new File(directory).mkdirs();
157                try {
158                        writer = new BufferedWriter(new FileWriter(directory
159                                        + "/ldp-testsuite-execution-report.html"));
160                        writer.write(output);
161
162                } catch (IOException e) {
163                } finally {
164                        try {
165                                if (writer != null)
166                                        writer.close();
167                        } catch (IOException e) {
168                        }
169                }
170        }

172        private void generateOverallSummaryReport(List<ISuite> suites, String id)
173                        throws IOException {
174                html.table(class_("summary"));
175                Date date = new Date();
176                for (ISuite suite : suites) {
177                        // Getting the results for the said suite
178                        Map<String, ISuiteResult> suiteResults = suite.getResults();
179
180                        for (ISuiteResult sr : suiteResults.values()) {
181
182                                ITestContext tc = sr.getTestContext();
183
184                                Iterator<ITestResult> passedResults = tc.getPassedTests()
185                                                .getAllResults().iterator();
186                                while (passedResults.hasNext()) {
187                                        ITestResult result = passedResults.next();
188
189                                        try {
190                                                Method m = result.getMethod().getConstructorOrMethod()
191                                                                .getMethod();
192                                                String[] groups = m.getAnnotation(Test.class).groups();
193                                                for (int i = 0; i < groups.length; i++) {
194                                                        if (groups[i].equals("MUST"))
195                                                                mustPass++;
196                                                        if (groups[i].equals("MAY"))
197                                                                mayPass++;
198                                                        if (groups[i].equals("SHOULD"))
199                                                                shouldPass++;
200
201                                                }
202                                        } catch (SecurityException e) {
203                                                e.printStackTrace();
204                                        }
205
206                                }
207                                Iterator<ITestResult> failedResults = tc.getFailedTests()
208                                                .getAllResults().iterator();
209                                while (failedResults.hasNext()) {
210                                        ITestResult result = failedResults.next();
211
212                                        try {
213                                                Method m = result.getMethod().getConstructorOrMethod()
214                                                                .getMethod();
215                                                String[] groups = m.getAnnotation(Test.class).groups();
216                                                for (int i = 0; i < groups.length; i++) {
217                                                        if (groups[i].equals("MUST"))
218                                                                mustFailed++;
219                                                        if (groups[i].equals("SHOULD"))
220                                                                shouldFailed++;
221                                                        if (groups[i].equals("MAY"))
222                                                                mayFailed++;
223                                                }
224                                        } catch (SecurityException e) {
225                                                e.printStackTrace();
226                                        }
227
228                                }
229                                Iterator<ITestResult> skippedResults = tc.getSkippedTests()
230                                                .getAllResults().iterator();
231                                while (skippedResults.hasNext()) {
232                                        ITestResult result = skippedResults.next();
233
234                                        try {
235                                                Method m = result.getMethod().getConstructorOrMethod()
236                                                                .getMethod();
237                                                String[] groups = m.getAnnotation(Test.class).groups();
238                                                for (int i = 0; i < groups.length; i++) {
239                                                        if (groups[i].equals("MUST"))
240                                                                mustSkip++;
241                                                        if (groups[i].equals("MAY"))
242                                                                maySkip++;
243                                                        if (groups[i].equals("SHOULD"))
244                                                                shouldSkip++;
245
246                                                }
247                                        } catch (SecurityException e) {
248                                                e.printStackTrace();
249                                        }
250
251                                }
252                                passed = tc.getPassedTests().getAllResults().size();
253                                failed = tc.getFailedTests().getAllResults().size();
254                                skipped = tc.getSkippedTests().getAllResults().size();
255                        }
256                        total = passed + failed + skipped;
257                        generateSummaryTableStart(date, suite.getName());
258                        generateSummaryTable();
259                        html._table();
260                }
261
262        }

264        private void generateSummaryTableStart(Date date, String suiteName)
265                        throws IOException {
266                html.tr().th().content("Test Suite Name");
267                html.th().content("Revision");
268                html.th().content("Report Date");
269                html.th().content("Skipped Tests");
270                html.th().content("Passed (MUST)");
271                html.th().content("Passed (MAY)");
272                html.th().content("Passed (SHOULD)");
273                // html.th().content("Failed (MUST)");
274                html._tr();
275
276                html.tr(class_("alt")).td().content(suiteName);
277                final String commit = BuildProperties.getRevision();
278                if (commit == null) {
279                        html.td().content("<UNKNOWN>");
280                } else {
281                        html.td().a(href("https://github.com/w3c/ldp-testsuite/commit/" + commit)).content(commit)._td();
282                }
283                html.td().content(date.toString());
284        }

286        private void generateSummaryTable()
287                        throws IOException {
288                printCellResult(skipped, total, "");
289                printCellResult(mustPass, mustPass + mustFailed, "MUST");
290                printCellResult(mayPass, mayPass + mayFailed, "SHOULD");
291                printCellResult(shouldPass, shouldPass + shouldFailed, "MAY");
292                html._tr();
293        }

295        private void printCellResult(int value, int total, String requirement) throws IOException {
296                if (value == 0)
297                        html.td().i().write("No tests of this type called")._i()._td();
298                else {
299                        DecimalFormat df = new DecimalFormat("##.##");
300                        double result = (((double) value / total) * 100);
301                        String finalTotal = df.format(result);
302                        html.td().b().write(finalTotal + "% ")._b();
303                        if(requirement.equals(""))
304                                html.write("of the total tests (");
305                        else 
306                                html.write(requirement + " Requirements (");
307                        
308                        html.b().write(value + "/" + total)._b();
309                        html.write(")")._td();
310                }
311        }

313        private void displayGroupsInfo(List<ISuite> suite) throws IOException {
314                for (ISuite testSuite : suite) {
315                        Map<String, ISuiteResult> tests = testSuite.getResults();
316                        for (ISuiteResult results : tests.values()) {
317                                ITestContext overview = results.getTestContext();
318                                String[] excluded = overview.getExcludedGroups();
319                                String[] included = overview.getIncludedGroups();
320                                generateList(included,
321                                                "Included Groups for " + testSuite.getName());
322                                generateList(excluded,
323                                                "Excluded Groups for " + testSuite.getName());
324                        }
325                }
326        }

353        private void makeMethodsList(ITestContext testContext) throws IOException {
354                IResultMap failed = testContext.getFailedTests();
355                IResultMap passed = testContext.getPassedTests();
356                IResultMap skipped = testContext.getSkippedTests();
357
358                html.h1(class_("center")).content("Methods called");
359                html.a(href("#Skipped")).write("Go To Skipped Tests").br()._a();
360                html.a(href("#Passed")).write("Go To Passed Tests").br()._a();
361                html.br();
362                
363                makeMethodSummaryTable(failed, "Failed");
364                html.br();
365                toTop();
366                html.br();
367                makeMethodSummaryTable(skipped, "Skipped");
368                html.br();
369                toTop();
370                html.br();
371                makeMethodSummaryTable(passed, "Passed");
372                html.br();
373
374        }

376        private void makeMethodSummaryTable(IResultMap tests, String title)
377                        throws IOException {
378                html.table(class_("indented"));
379                html.tr().th(class_(title)).a(id((title))).write(title + " Test Cases")._a()._th();
380                html.th(class_(title)).content("Groups");
381                html.th(class_(title)).content("Test Class");
382                html.th(class_(title)).content("Description of Test Method")._tr();
383                for (ITestResult result : tests.getAllResults()) {
384                        ITestNGMethod method = result.getMethod();
385                        List<String> groups = Arrays.asList(method.getGroups());
386                        if (groups.contains("MUST") && result.getStatus() == ITestResult.FAILURE) {
387                                html.tr(class_("critical"));
388                        } else {
389                                html.tr();
390                        }
391
392                        html.td()
393                                        .a(href("#" + method.getTestClass().getName() + "_"
394                                                        + method.getMethodName()))
395                                        .write(method.getMethodName(), NO_ESCAPE)._a()._td();
396                        html.td().content(Arrays.toString(method.getGroups()));
397                        html.td().content(method.getTestClass().getName());
398
399                        html.td().content(
400                                        (method.getDescription() != null ? method.getDescription()
401                                                        : "No Description found"));
402                        html._tr();
403                }
404                html._table();
405        }

407        private void generateMethodDetails(List<ISuite> suites) throws IOException {
408                html.h1().content("Test Method Details");
409                for (ISuite suite : suites) {
410                        Map<String, ISuiteResult> r = suite.getResults();
411                        for (ISuiteResult r2 : r.values()) {
412                                ITestContext testContext = r2.getTestContext();
413
414                                generateDetail(testContext.getFailedTests());
415                                generateDetail(testContext.getSkippedTests());
416                                generateDetail(testContext.getPassedTests());
417                        }
418                }
419        }

421        private void generateDetail(IResultMap tests) throws IOException {
422                for (ITestResult m : tests.getAllResults()) {
423                        ITestNGMethod method = m.getMethod();
424                        html.h2()
425                                        .a(id(m.getTestClass().getName() + "_"
426                                                        + method.getMethodName()))
427                                        .write(m.getTestClass().getName() + ": "
428                                                        + method.getMethodName())._a()._h2();
429                        getAdditionalInfo(m, method);
430                        html.p(class_("indented"))
431                                        .b()
432                                        .write("Description: ")
433                                        ._b()
434                                        .write((method.getDescription() != null ? method
435                                                        .getDescription()
436                                                        : "No description for this test method found"))
437                                        ._p();
438                        String groups = "";
439                        for (String group : method.getGroups()) {
440                                groups += group + " ";
441                        }
442                        html.p(class_("indented")).b().write("Requirement Level: ")._b()
443                                        .write(groups)._p();
444
445                        toTop();
446                }
447
448        }

450        private void getAdditionalInfo(ITestResult m, ITestNGMethod method)
451                        throws IOException {
452                if (m.getThrowable() != null) {
453                        Throwable thrown = m.getThrowable();
454                        String exception = thrown.getClass().getName();
455                        if (exception.contains("Skip")) {
456                                createSkipExceptionTable(thrown);
457                        } else {
458                                createThrownTable(thrown);
459                        }
460                }
461
462                if (m.getParameters() != null && m.getParameters().length != 0) {
463                        Object[] params = m.getParameters();
464                        String parameters = "";
465                        for (Object p : params) {
466                                if (p != null)
467                                        parameters += p.toString() + " ";
468                        }
469                        html.p(class_("indented")).b().write("Parameters: ")._b()
470                                        .write(parameters)._p();
471                }
472
473                String reference = "";
474                if (m.getMethod().getConstructorOrMethod().getMethod()
475                                .getAnnotation(SpecTest.class) != null) {
476                        reference = m.getMethod().getConstructorOrMethod().getMethod()
477                                        .getAnnotation(SpecTest.class).specRefUri();
478                        html.p(class_("indented")).b().write("Reference URI: ")._b()
479                                        .a(href(reference)).write(reference)._a()._p();
480                }
481
482        }

562                Iterator<String> label = names.iterator();
563                int textY = 13;
564                int rectY = 0;

568                        String className = label.next();

570                                        + "\" style=\"fill:" + colors.get(getColor) + "\"/>",

573                                        + className + "</text>", NO_ESCAPE);
574
575                        textY += 20;
576                        rectY += 20;
577                        getColor++;
578                
579                }
580                html.write("</svg>");
581        }

583        private void writeChartLabels(HashMap<String, Integer> passClasses,
584                        HashMap<String, Integer> failClasses,
585                        HashMap<String, Integer> skipClasses) {
586                graphs.write("datalabels: {");
587                Set<String> names = passClasses.keySet();
588                Iterator<String> label = names.iterator();
589                while (label.hasNext()) {
590                        String className = label.next();
591                        graphs.write(className + ": [ \"" + passClasses.get(className)
592                                        + " " + className + "\", \"" + failClasses.get(className)
593                                        + " " + className + "\", \"" + skipClasses.get(className)
594                                        + " " + className + "\" ]");
595                        if (label.hasNext())
596                                graphs.write(",");
597                }
598                graphs.write("},");
599        }

601        private void writeChartColors(HashMap<String, Integer> passClasses) {
602                graphs.write("colors: {");
603                Set<String> names = passClasses.keySet();
604                Iterator<String> label = names.iterator();
605                int getColor = 0;
606                while (label.hasNext() && getColor != colors.size()) {
607                        String className = label.next();
608
609                        if (label.hasNext())
610                                graphs.write(className + ": '" + colors.get(getColor) + "', ");
611                        else
612                                graphs.write(className+ ": '" + colors.get(getColor) + "' },");
613                        getColor++;
614                }
615        }

633        }

635        private HashMap<String, Integer> getClasses(IResultMap tests) {
636                HashMap<String, Integer> classes = new HashMap<String, Integer>();
637                Iterator<ITestResult> results = tests.getAllResults().iterator();
638                while (results.hasNext()) {
639                        String name = results.next().getTestClass().getName().toString();
640                        name = name.substring(name.lastIndexOf(".") + 1);
641
642                        if (!classes.containsKey(name))
643                                classes.put(name, 1);
644                        else
645                                classes.put(name, classes.get(name) + 1);
646                }
647                return classes;
648        }
