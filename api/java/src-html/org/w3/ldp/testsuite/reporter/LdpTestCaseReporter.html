342        private static void iterateTableList(HashMap<String, String> list) throws IOException {
343                String className;
344                String methodName;
345                Iterator<Entry<String, String>> codeIter = list.entrySet().iterator();
346                while(codeIter.hasNext()){
347                        Entry<String, String> entry = codeIter.next();
348                        className = entry.getValue();
349                        className = className.substring(className.lastIndexOf(".") + 1);
350                        methodName = entry.getKey();
351                        html.li().a(href("#" + methodName)).b().write(className + "::")._b().write(methodName)._a()._li();
352                }
353        }

371        private static <T> void acquireTestInfo(Class<T> classType)
372                        throws IOException {
373                int total = 0, must = 0, should = 0, may = 0;
374                
375                int[] autoReq = { 0, 0, 0 }, unimReq = { 0, 0, 0 }, clientReq = { 0, 0, 0 }, manReq = { 0, 0, 0 };
376                int[] apprReq = { 0, 0, 0 }, pendReq = { 0, 0, 0 }, extReq = { 0, 0, 0 }, 
377                                depreReq = {0, 0, 0 }, clariReq = { 0, 0, 0 };
378
379                Method[] methods = classType.getDeclaredMethods();
380
381                for (Method method : methods) {
382                        if (method.isAnnotationPresent(SpecTest.class)
383                                        && method.isAnnotationPresent(Test.class)) {
384                                total++;
385                                SpecTest testLdp = method.getAnnotation(SpecTest.class);
386                                Test test = method.getAnnotation(Test.class);
387                                String group = Arrays.toString(test.groups());
388                                if (group.contains("MUST"))
389                                        must++;
390                                if (group.contains("SHOULD"))
391                                        should++;
392                                if (group.contains("MAY"))
393                                        may++;
394                                switch (testLdp.testMethod()) {
395                                case AUTOMATED:
396                                        if (group.contains("MUST"))
397                                                ++autoReq[MUST];
398                                        if (group.contains("SHOULD"))
399                                                ++autoReq[SHOULD];
400                                        if (group.contains("MAY"))
401                                                ++autoReq[MAY];
402                                        break;
403                                case NOT_IMPLEMENTED:
404                                        if (group.contains("MUST"))
405                                                ++unimReq[MUST];
406                                        if (group.contains("SHOULD"))
407                                                ++unimReq[SHOULD];
408                                        if (group.contains("MAY"))
409                                                ++unimReq[MAY];
410                                        break;
411                                case MANUAL:
412                                        if (group.contains("MUST"))
413                                                ++manReq[MUST];
414                                        if (group.contains("SHOULD"))
415                                                ++manReq[SHOULD];
416                                        if (group.contains("MAY"))
417                                                ++manReq[MAY];
418                                        break;
419                                case CLIENT_ONLY:
420                                        if (group.contains("MUST"))
421                                                ++clientReq[MUST];
422                                        if (group.contains("SHOULD"))
423                                                ++clientReq[SHOULD];
424                                        if (group.contains("MAY"))
425                                                ++clientReq[MAY];
426                                        break;
427                                default:
428                                        break;
429                                }
430                                switch (testLdp.approval()) {
431                                case WG_PENDING:
432                                        if (group.contains("MUST"))
433                                                ++pendReq[MUST];
434                                        if (group.contains("SHOULD"))
435                                                ++pendReq[SHOULD];
436                                        if (group.contains("MAY"))
437                                                ++pendReq[MAY];
438                                        break;
439                                case WG_APPROVED:
440                                        if (group.contains("MUST"))
441                                                ++apprReq[MUST];
442                                        if (group.contains("SHOULD"))
443                                                ++apprReq[SHOULD];
444                                        if (group.contains("MAY"))
445                                                ++apprReq[MAY];
446                                        break;
447                                case WG_EXTENSION:
448                                        if (group.contains("MUST"))
449                                                ++extReq[MUST];
450                                        if (group.contains("SHOULD"))
451                                                ++extReq[SHOULD];
452                                        if (group.contains("MAY"))
453                                                ++extReq[MAY];
454                                        break;
455                                case WG_DEPRECATED:
456                                        if (group.contains("MUST"))
457                                                ++depreReq[MUST];
458                                        if (group.contains("SHOULD"))
459                                                ++depreReq[SHOULD];
460                                        if (group.contains("MAY"))
461                                                ++depreReq[MAY];
462                                        break;
463                                case WG_CLARIFICATION:
464                                        if (group.contains("MUST"))
465                                                ++clariReq[MUST];
466                                        if (group.contains("SHOULD"))
467                                                ++clariReq[SHOULD];
468                                        if (group.contains("MAY"))
469                                                ++clariReq[MAY];
470                                        break;
471                                default:
472                                        break;
473                                }
474                        }
475                }
476
477                // write information into Grafico bar charts
478                writeStatusGraph(classType.getSimpleName(), apprReq, pendReq, extReq, depreReq, clariReq);
479                writeImplementationGraph(classType.getSimpleName(), autoReq, unimReq, clientReq, manReq);
480
481                html.table(class_("classes"));
482
483                html.tr().th().content("Total Tests");
484                html.th().content("Coverage");
485                html.th().content("Status");
486                html.th().content("Implementation");
487                html._tr();
488
489                html.tr().td().content(total + "");
490                html.td();
491                if (must > 0)
492                        html.b().write("MUST: ")._b().write(must).br();
493                if (should > 0)
494                        html.b().write("SHOULD: ")._b().write(should).br();
495                if (may > 0)
496                        html.b().write("MAY: ")._b().write(may).br();
497
498                html._td().td();
499                int approve = getTotal(apprReq);
500                if(approve > 0)
501                        html.b().write("WG Approved: ")._b().write("" + approve).br();
502                int pend = getTotal(pendReq);
503                if(pend > 0)
504                        html.b().write("Pending: ")._b().write("" + pend).br();
505                int extend = getTotal(extReq);
506                if (extend > 0)
507                        html.b().write("Extension: ")._b().write("" + extend).br();
508                int depre = getTotal(depreReq);
509                if (depre > 0)
510                        html.b().write("Deprecated: ")._b().write("" + depre).br();
511                int clarify = getTotal(clariReq);
512                if(clarify > 0)
513                        html.b().write("Clarification: ")._b().write("" + clarify).br();                
514
515                html._td().td();
516                int auto = getTotal(autoReq);
517                if (auto > 0)
518                        html.b().write("Automated: ")._b().write("" + auto).br();
519                int unimpl = getTotal(unimReq);
520                if (unimpl > 0)
521                        html.b().write("Not Implemented: ")._b().write("" + unimpl).br();
522                int client = getTotal(clientReq);
523                if (client > 0)
524                        html.b().write("Client Only: ")._b().write("" + client).br();
525                int manual = getTotal(manReq);
526                if (manual > 0)
527                        html.b().write("Manual: ")._b().write("" + manual).br();
528
529                html._td();
530
531                html._tr();
532                html._table();
533
534        }

615        private static <T> void acquireTestCases(Class<T> classType)
616                        throws IOException {
617                String name = classType.getCanonicalName();
618                if (initialRead)
619                        html.h2().a(id(name)).write("Test Class: " + name)._a()._h2();
620                Method[] declaredMethods = classType.getDeclaredMethods();
621
622                if (initialRead)
623                        html.ul();
624                for (Method method : declaredMethods) {
625                        if (method.isAnnotationPresent(Test.class)) {
626                                if (!initialRead) {
627                                        generateInformation(method, name);
628                                } else {
629                                        html.li(id(method.getName()));
630                                        html.b();
631                                        html.a(href(ReportUtils.getJavadocLink(method)))
632                                                        .content(method.getName());
633                                        html._b();
634                                        generateInformation(method, name);
635                                        html._li();
636                                }
637                        }
638                }
639                if (initialRead) {
640                        html._ul();
641                        toTop();
642                }
643        }

645        private static void generateInformation(Method method, String name)
646                        throws IOException {
647                SpecTest testLdp = null;
648                Test test = null;
649                if (method.getAnnotation(SpecTest.class) != null
650                                && method.getAnnotation(Test.class) != null) {
651                        testLdp = method.getAnnotation(SpecTest.class);
652                        test = method.getAnnotation(Test.class);
653
654                        if (!initialRead) {
655                                totalTests++;
656                                METHOD methodStatus = testLdp.testMethod();
657                                STATUS testApproval = testLdp.approval();
658                                String group = Arrays.toString(test.groups());
659                                if (!refURI.contains(testLdp.specRefUri())) { // for just the requirement testing
660                                        refURI.add(testLdp.specRefUri());
661                                        if (group.contains("MUST")) {
662                                                mustTotal++;
663                                                switch (methodStatus) {
664                                                case AUTOMATED:
665                                                        automated++;

667                                                        if (testApproval.equals(STATUS.WG_PENDING))
668                                                                readyToBeApproved.put(method.getName(), method.getDeclaringClass().getCanonicalName());
669                                                        break;

681                                                        break;
682                                                default:
683                                                        break;
684                                                }
685                                                switch (testApproval) {
686                                                case WG_APPROVED:
687                                                        ++approve[MUST];
688                                                        approved++;
689                                                        break;
690                                                case WG_CLARIFICATION:
