001package org.w3.ldp.testsuite.reporter;
002
003import java.io.BufferedWriter;
004import java.io.File;
005import java.io.FileWriter;
006import java.io.IOException;
007import java.io.StringWriter;
008import java.util.HashMap;
009import java.util.Map.Entry;
010
011import org.w3.ldp.testsuite.vocab.LDP;
012
013import com.github.jsonldjava.core.JsonLdError;
014import com.github.jsonldjava.core.JsonLdOptions;
015import com.github.jsonldjava.core.JsonLdProcessor;
016import com.github.jsonldjava.jena.JenaJSONLD;
017import com.github.jsonldjava.utils.JsonUtils;
018import com.hp.hpl.jena.rdf.model.Model;
019import com.hp.hpl.jena.rdf.model.ModelFactory;
020
021public abstract class AbstractEarlReporter {
022
023        protected BufferedWriter writerTurtle;
024        protected BufferedWriter writerJson;
025        protected Model model;
026        protected static final String TURTLE = "TURTLE";
027        protected static final String JSON_LD = "JSON-LD";
028        protected static final String OUTPUT_DIR = "report";
029        protected static final HashMap<String, String> prefixes = new HashMap<String, String>();
030
031        static {
032                JenaJSONLD.init();
033
034                prefixes.put("doap", "http://usefulinc.com/ns/doap#");
035                prefixes.put("foaf", "http://xmlns.com/foaf/0.1/");
036                prefixes.put("earl", "http://www.w3.org/ns/earl#");
037                prefixes.put("rdf", "http://www.w3.org/1999/02/22-rdf-syntax-ns#");
038                prefixes.put("rdfs", "http://www.w3.org/2000/01/rdf-schema#");
039                prefixes.put("mf", "http://www.w3.org/2001/sw/DataAccess/tests/test-manifest#");
040                prefixes.put("rdft", "http://www.w3.org/ns/rdftest#");
041                prefixes.put("dcterms", "http://purl.org/dc/terms/");
042                prefixes.put("td", "http://www.w3.org/2006/03/test-description#");
043                prefixes.put(LDP.LDPT_PREFIX, LDP.LDPT_NAMESPACE);
044        }
045
046        protected abstract String getFilename();
047
048        protected void createWriter(String directory) throws IOException {
049                File dir = new File(directory);
050                dir.mkdirs();
051                writerTurtle = new BufferedWriter(new FileWriter(new File(dir, getFilename() + ".ttl")));
052                writerJson = new BufferedWriter(new FileWriter(new File(dir, getFilename() + ".jsonld")));
053        }
054
055        protected void write() {
056                model.write(writerTurtle, TURTLE);
057
058                StringWriter sw = new StringWriter();
059                model.write(sw, JSON_LD);
060                try {
061                        Object jsonObject = JsonUtils.fromString(sw.toString());
062                        Object compact = JsonLdProcessor.compact(jsonObject, prefixes, new JsonLdOptions());
063                        writerJson.write(JsonUtils.toPrettyString(compact));
064                } catch (IOException | JsonLdError e) {
065                        e.printStackTrace();
066                }
067        }
068
069        protected void endWriter() throws IOException {
070                writerTurtle.flush();
071                writerTurtle.close();
072
073                writerJson.flush();
074                writerJson.close();
075        }
076
077        protected void createModel() {
078                model = ModelFactory.createDefaultModel();
079                writePrefixes(model);
080        }
081        
082        public void writePrefixes(Model model) {
083                for (Entry<String, String> prefix : prefixes.entrySet()) {
084                        model.setNsPrefix(prefix.getKey(), prefix.getValue());
085                }
086        }
087
088        public String createTestCaseName(String className, String methodName) {
089
090                className = className.substring(className.lastIndexOf(".") + 1);
091
092                if (className.endsWith("Test")) {
093                        className = className.substring(0, className.length()-4);
094                }
095                if (methodName.startsWith("test")) {
096                        methodName = methodName.substring(4, methodName.length());
097                }
098                return className + "-" + methodName;
099        }
100
101        public String createTestCaseURL(String className, String methodName) {
102                return LDP.LDPT_NAMESPACE + createTestCaseName(className, methodName);
103        }
104}
