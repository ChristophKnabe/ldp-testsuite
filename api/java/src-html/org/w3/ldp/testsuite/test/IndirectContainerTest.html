039        @Test(
040                        groups = {MUST},
041                        description = "LDP servers exposing LDPCs MUST advertise their "
042                                        + "LDP support by exposing a HTTP Link header with a "
043                                        + "target URI matching the type of container (see below) "
044                                        + "the server supports, and a link relation type of type "
045                                        + "(that is, rel='type') in all responses to requests made "
046                                        + "to the LDPC's HTTP Request-URI.")
047        @SpecTest(
048                        specRefUri = LdpTestSuite.SPEC_URI + "#ldpc-linktypehdr",
049                        testMethod = METHOD.AUTOMATED,
050                        approval = STATUS.WG_APPROVED)
051        public void testContainerSupportsHttpLinkHeader() {
052                Response response = buildBaseRequestSpecification().header(ACCEPT, TEXT_TURTLE)
053                                .expect().statusCode(HttpStatus.SC_OK).when()
054                                .get(indirectContainer);
055                assertTrue(
056                                containsLinkHeader(LDP.IndirectContainer.stringValue(), LINK_REL_TYPE,
057                                                response),
058                                "LDP DirectContainers must advertise their LDP support by exposing a HTTP Link header with a URI matching <"
059                                                + LDP.IndirectContainer.stringValue()
060                                                + "> and rel='type'"
061                );
062        }
