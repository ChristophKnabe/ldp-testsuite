001package org.w3.ldp.testsuite.test;
002
003import static org.hamcrest.Matchers.notNullValue;
004
005import java.io.IOException;
006
007import org.apache.http.HttpStatus;
008import org.testng.SkipException;
009import org.testng.annotations.AfterSuite;
010import org.testng.annotations.BeforeSuite;
011import org.testng.annotations.Optional;
012import org.testng.annotations.Parameters;
013import org.w3.ldp.testsuite.mapper.RdfObjectMapper;
014
015import com.hp.hpl.jena.rdf.model.Model;
016import com.jayway.restassured.response.Response;
017
018/**
019 * Tests that run on an LDP-RS that is not a container.
020 */
021public class MemberResourceTest extends RdfSourceTest {
022
023        private String container;
024        private String memberResource;
025
026        @Parameters("auth")
027        public MemberResourceTest(@Optional String auth) throws IOException {
028                super(auth);
029        }
030
031        @Parameters({"memberResource", "directContainer", "indirectContainer", "basicContainer", "memberTtl"})
032        @BeforeSuite(alwaysRun = true)
033        public void setup(@Optional String memberResource, @Optional String directContainer,
034                        @Optional String indirectContainer, @Optional String basicContainer,
035                        @Optional String memberTtl) {
036                // If resource is defined, use that. Otherwise, fall back to creating one from one of the containers.
037                if (memberResource != null) {
038                        this.memberResource = memberResource;
039                } else if (directContainer != null) {
040                        this.container = directContainer;
041                } else if (indirectContainer != null) {
042                        this.container = indirectContainer;
043                } else if (basicContainer != null) {
044                        this.container = basicContainer;
045                } else {
046                        throw new SkipException("No memberResource or container parameters defined in testng.xml");
047                }
048
049                if (this.memberResource == null) {
050                        try {
051                                Model model = this.readModel(memberTtl);
052                                if (model == null) {
053                                        model = this.getDefaultModel();
054                                }
055
056                                Response postResponse = buildBaseRequestSpecification()
057                                                .contentType(TEXT_TURTLE)
058                                                        .body(model, new RdfObjectMapper())
059                                                .expect()
060                                                        .statusCode(HttpStatus.SC_CREATED)
061                                                        .header(LOCATION, notNullValue())
062                                                .when()
063                                                        .post(this.container);
064
065                                this.memberResource = postResponse.getHeader(LOCATION);
066                        } catch (Exception e) {
067                                System.err.println("ERROR: Could not create test resource for MemberResourceTest. Skipping tests.");
068                                e.printStackTrace();
069                        }
070                }
071        }
072
073        @Override
074        protected String getResourceUri() {
075                if (memberResource == null) {
076                        throw new SkipException("Skipping test because test resource is null.");
077                }
078                return memberResource;
079        }
080
081        @AfterSuite(alwaysRun = true)
082        public void tearDown() {
083                // If container isn't null, we created the resource ourselves. To clean up, delete the resource.
084                if (container != null) {
085                        buildBaseRequestSpecification().delete(memberResource);
086                }
087        }
088}
