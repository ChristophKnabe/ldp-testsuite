049        @BeforeClass(alwaysRun = true)
050        public void determineOptions() {
051                String uri = getResourceUri();
052                if (StringUtils.isNotBlank(uri)) {
053                        // Use HTTP OPTIONS, which MUST be supported by LDP servers, to determine what methods are supported on this container.
054                        Response optionsResponse = buildBaseRequestSpecification().options(uri);
055                        String allow = optionsResponse.header(ALLOW);
056                        if (allow != null) {
057                                String[] methods = allow.split("\\s*,\\s*");
058                                for (String method : methods) {
059                                        options.add(method);
060                                }
061                        }
062                }
063        }
064
065        @Parameters("auth")
066        public CommonResourceTest(@Optional String auth) throws IOException {
067                if (StringUtils.isNotBlank(auth) && auth.contains(":")) {
068                        String[] split = auth.split(":");
069                        if (split.length == 2 && StringUtils.isNotBlank(split[0]) && StringUtils.isNotBlank(split[1])) {
070                                this.auth = ImmutableMap.of("username", split[0], "password", split[1]);
071                        }
072                } else {
073                        this.auth = null;
074                }
075        }
076
077        @Override
078        protected RequestSpecification buildBaseRequestSpecification() {
079                if (auth == null) {
080                        return RestAssured.given();
081                } else {
082                        return RestAssured.given().auth().preemptive().basic(auth.get("username"), auth.get("password"));
083                }
084        }

277        @Test(
278                        groups = {MUST},
279                        description = "LDP servers MUST respond with status code 412 "
280                                        + "(Condition Failed) if ETags fail to match when there "
281                                        + "are no other errors with the request [HTTP11]. LDP "
282                                        + "servers that require conditional requests MUST respond "
283                                        + "with status code 428 (Precondition Required) when the "
284                                        + "absence of a precondition is the only reason for rejecting "
285                                        + "the request [RFC6585].")
286        @SpecTest(
287                        specRefUri = LdpTestSuite.SPEC_URI + "#ldpr-put-precond",
288                        testMethod = METHOD.AUTOMATED,
289                        approval = STATUS.WG_APPROVED)
290        public void testPreconditionRequiredStatusCode() {
291                skipIfMethodNotAllowed(HttpMethod.PUT);
292
293                String resourceUri = getResourceUri();
294                Response getResponse = buildBaseRequestSpecification()
295                                .expect()
296                                .statusCode(isSuccessful())
297                                .header(ETAG, isValidEntityTag())
298                                .when()
299                                .get(resourceUri);
300
301                // Verify that we can successfully PUT the resource WITH an If-Match header.
302                Response ifMatchResponse = buildBaseRequestSpecification()
303                                        .header(IF_MATCH, getResponse.getHeader(ETAG))
304                                        .contentType(getResponse.contentType())
305                                        .body(getResponse.asByteArray())
306                                .when()
307                                        .put(resourceUri);
308                if (!isSuccessful().matches(ifMatchResponse.getStatusCode())) {
309                        throw new SkipException("Skipping test because PUT request failed with valid If-Match header.");
310                }
311
312                // Now try WITHOUT the If-Match header. If the result is NOT successful,
313                // it should be because the header is missing and we can check the error
314                // code.
315                Response noIfMatchResponse = buildBaseRequestSpecification()
316                                        .contentType(getResponse.contentType())
317                                        .body(getResponse.asByteArray())
318                                .when()
319                                        .put(resourceUri);
320                if (isSuccessful().matches(noIfMatchResponse.getStatusCode())) {
321                        // It worked. This server doesn't require If-Match, which is only a
322                        // SHOULD requirement (see testPutRequiresIfMatch). Skip the test.
323                        throw new SkipException("Server does not require If-Match header.");
324                }
325
326                assertEquals(428, noIfMatchResponse.getStatusCode(), "Expected 428 Precondition Required error on PUT request with no If-Match header");
327        }

419        protected boolean supports(HttpMethod method) {
420                return options.contains(method.getName());
