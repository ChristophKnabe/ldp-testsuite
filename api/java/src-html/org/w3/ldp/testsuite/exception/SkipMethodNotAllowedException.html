001package org.w3.ldp.testsuite.exception;
002
003import org.testng.SkipException;
004import org.w3.ldp.testsuite.http.HttpMethod;
005
006public class SkipMethodNotAllowedException extends SkipException {
007        private static final long serialVersionUID = 1L;
008
009        public SkipMethodNotAllowedException(HttpMethod method) {
010                super("Skipping test since method " + method.getName() +
011                                " is not allowed. This HTTP method is needed for this test.");
012        }
013
014        public SkipMethodNotAllowedException(String uri, HttpMethod method) {
015                super("Skipping test since <" + uri + "> has not advertised " + method.getName() +
016                                " support through its HTTP OPTIONS response. This HTTP method is needed for this test.");
017        }
018}
